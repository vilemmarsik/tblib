  14. /**
  15.   * Function to process GROUP BY and ORDER BY column specs
  16.   * @param $cols array of columns
  17.   * @param $by_nr column number(s) counted from 1, negative to append ' DESC'
  18.   * @param $by_name column names (or any random description, this is not validated)
  19.   * @return array of selected columns
  20.   ***/
  21. function _select_columns($cols,$by_nr,$by_name)
  22. {
  23.     $ret=array();
  24.     if( $by_nr )    {
  25.         if( !is_array($by_nr) )
  26.             $by_nr=array($by_nr);
  27.         foreach( $by_nr as $nr )
  28.             $ret[]=$cols[abs($nr)-1].($nr<0 ? ' DESC':'');
  29.     }
  30.     if( $by_name )    {
  31.         if( is_array($by_name) )
  32.             $ret=array_merge($ret,$by_name);
  33.         else
  34.             $ret[]=$by_name;
  35.     }
  36.     return $ret;
  37. }

  39. /**
  40.   * Common search code.
  41.   * Builds a search query, executes, returns the results.
  42.   * @param array $sel array of attributes to select
  43.   * @param string $from string containing the join to search from
  44.   * @param array &$attrs requested search values,
  45.   *    in form of array( 'key1,key1,...'=>array(val1,val2,...), ... ).
  46.   *    Specifying multiple values means that all combinations are made, and ORed.
  47.   * @param array &$attrs_known array of array(<SQL condition>, <bind format letter>)
  48.   * @return array 2D array with search results
  49.   ***/
  50. function search_common( $sel, $from, &$attrs, &$attrs_known, &$pager=null )
  51. {
  52.     global $first;
  53.     $header   = hash_get($attrs,'header',true,true);    # return table header ?
  54.     $limit    = hash_get($attrs,'limit',null,true);        # row nr. limit (array)
  55.     $order_nr = hash_get($attrs,'order_nr',0,true);        # index of ORDER BY columns
  56.     $order_by = hash_get($attrs,'order_by',null,true);    # columns for ORDER BY
  57.     $page     = hash_get($attrs,'page',$first,true);    # page number
  58.     $perpage  = hash_get($attrs,'perpage',null,true);    # rows per page
  59.     $where    = hash_get($attrs,'where',null,true);        # starting WHERE clause
  60.     $just_sql = hash_get($attrs,'just_sql',false,true);    # return SQL instead of results ?
  61.     $group_nr = hash_get($attrs,'group_nr',null,true);    # index of GROUP BY columns
  62.     $group_by = hash_get($attrs,'group_by',null,true);    # columns for GROUP BY
  63.  
  64.     $select=is_array($sel) ? join(', ',$sel) : $sel;
  65.     $sql="FROM $from WHERE 1";
  66.  
  67.     if( $where )
  68.         $sql .= " AND $where";
  69.  
  70.     # ORDER BY / GROUP BY clauses
  71.     $order = _select_columns($sel,$order_nr,$order_by);
  72.     $group = _select_columns($sel,$group_nr,$group_by);
  73.     
  74.     # WHERE clause
  75.     $args=array();
  76.     $format='';
  77.     foreach($attrs as $keys=>$vals)
  78.     {    # AND from all the conditions received
  79.         if( !set($vals) ) continue;
  80.         if( is_array($vals) && ( count($vals)==0 || !set($vals[0]) ) )
  81.             continue;
  82.  
  83.         $keys=split(',',$keys);
  84.  
  85.         $sql_part=array();
  86.         foreach( $keys as $key )
  87.         {    # OR from all key=>val combinations
  88.  
  89.             # check correct calling
  90.             if( !isset($attrs_known[$key]) )
  91.                 abort("search_common: unknown attribute : $key");
  92.  
  93.             $a=$attrs_known[$key];
  94.             
  95.             # sanitize one-value arrays
  96.             if( !is_array($vals) )
  97.                 $vals = array($vals);
  98.  
  99.             foreach( $vals as $val )
  100.             {
  101.                 $sql_part[] = $a[0];
  102.                 $format    .= $a[1];
  103.                 $args[]    .= $val;
  104.             }
  105.         }
  106.         if( count($sql_part) == 1 )
  107.             $sql .= " AND ".$sql_part[0];
  108.         else if( count($sql_part) > 1 )
  109.         {
  110.             $sql .= " AND ( 0";
  111.             foreach( $sql_part as $s )
  112.                 $sql .= " OR $s";
  113.             $sql .= ")";
  114.         }
  115.     }
  116.     if( count($group) )
  117.         $sql .= ' GROUP BY '.join(',',$group);
  118.     if( count($order) )
  119.         $sql.=' ORDER BY '.join(',',$order);
  120. #    print("SQL : $sql<br/>\n");
  121. #    print("format : $format<br/>\n");
  122. #    print("Arguments : ");
  123. #    print_r($args);
  124.     array_unshift($args, $format);
  125.     
  126.     # query count
  127.     if( !is_null($pager) )
  128.     {    
  129. #        print "<pre>";print_r($pager);print "</pre>\n";
  130.         $args1 = $args;
  131.         array_unshift($args1,"SELECT COUNT(*) $sql");
  132.         $count = call_user_func_array('scalar_query', $args1);
  133.  
  134.         $limit = limit_from_pager($pager,$count);
  135.     }
  136.     if( !isset($limit) )
  137.         $limit = array(5000);
  138.     
  139.     # query data
  140.     array_unshift($args, $header, $limit, "SELECT $select $sql" );
  141.     return $just_sql ? $args : call_user_func_array('mhash_query', $args);
  142. }

  144. function limit_from_pager(&$pager, $count)
  145. {
  146.     global $first;
  147.     $limit = null;
  148.     $pager['count'] = $count;
  149.     if( isset($pager['page']) && isset($pager['rpp']) )    {
  150.         $page = max( 0, min( $pager['page']-$first, ceil($count/$pager['rpp'])-1 ) );
  151.         $limit = array( $page*$pager['rpp'], $pager['rpp'] );
  152.         $pager['page'] = $page+$first;
  153.     }
  154.     return $limit;
  155. }

  242. /** gets list of names, returns list of IDs, omits unknown names, removes known names from &$names */
  243. function enum_map_to_ids($tbl, &$names, $wildcard=0)
  244. {
  245.     $ret=array();
  246.     $c=count($names);
  247.     for( $i=0; $i<$c; $i++ )
  248.     {
  249.         if( $wildcard )
  250.         {
  251.             $ids=enum_get_id_wildcard($tbl,$names[$i]);
  252.             $ret=array_merge($ret,$ids);
  253.             if( $ids ) unset($names[$i]);
  254.         }
  255.         else
  256.         {
  257.             $id=enum_get_id($tbl,$names[$i]);
  258.             if( !is_null($id) )
  259.             {
  260.                 $ret[]=$id;
  261.                 unset($names[$i]);
  262.             }
  263.         }
  264.     }
  265.     return $ret;
  266. }

  400. /**

  432. /**

  441.     $args=func_get_args();
  442.     array_unshift($args,null);    // the LIMIT clause
  443.     $statement=call_user_func_array('cached_query',$args);
  444.     if( !$statement )
