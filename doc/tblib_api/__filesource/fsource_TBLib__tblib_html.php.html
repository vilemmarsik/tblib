  325. function html_groupped_table($data,$args=array())
  326. {
  327.     $args = args_defaults( $args, array(
  328.         'group_y'=>array(),        # Y attributes; !!! NOTE: $data must be sorted by group_y !!!
  329.         'group_x'=>array(),        # X attributes
  330.         'header'=>0,            # 1 to consider $data[0] to be a header
  331.         'aggregate_fields'=>array(),    # fields to aggregate into one
  332.         'aggregate_name'=>'summary',    # name of the resulting column
  333.         'aggregate_callback'=>null,    # callback to aggregate fields into one
  334.         'aggregate_arg'=>null,        # extra argument passed to aggregate_callback
  335.         'filter_callback'=>null,    # callback that can filter out some rows
  336.         'filter_arg'=>null        # extra argument passed to filter_callback
  337.     ));
  338.     # concatenated axis values, to be used as hash keys
  339.     $keys=array( 'y'=>array(), 'x'=>array() );
  340.  
  341.     # groupping attributes
  342.     $group=array( 'y'=>$args['group_y'], 'x'=>$args['group_x'], 'a'=>$args['aggregate_fields'] );
  343.  
  344.     # split the table data into parts
  345.     foreach( $data as $num=>$row )    {
  346.         $is_header = ( $num=='header' || ($num==0 && $args['header']) );
  347.  
  348.         # key is the concatenated value, we define it here and just reuse later
  349.         $key=array();
  350.  
  351.         foreach( array('y','x','a') as $i )    {
  352.  
  353.             # remove X/Y/aggregate fields from data
  354.             $val[$i] = array();
  355.             foreach( $group[$i] as $gkey )    {
  356.                 if( isset($row[$gkey]) )    {
  357.                     $val[$i][$gkey] = $row[$gkey];
  358.                     unset($row[$gkey]);
  359.                 }
  360.             }
  361.  
  362.             # for X and Y, concatenate & make a key
  363.             if( $i!='a' )    {
  364.                 $key[$i] = join('|',$val[$i]);
  365.                 if( !$is_header && !isset($keys[$i][$key[$i]]) )
  366.                     $keys[$i][$key[$i]] = $val[$i];
  367.             }
  368.         }
  369.  
  370.         # now replace the original data row
  371.         $data[$num] = array( 
  372.             'y_key'=>$key['y'],
  373.             'x_key'=>$key['x']
  374.         );
  375.         $data[$num]['data'] = $row;
  376.  
  377.         # aggregation is made here
  378.         if( $args['aggregate_callback'] && count($args['aggregate_fields']) )    {
  379.  
  380.             # data are prepared by the aggregation callback
  381.             $data[$num]['data'][$args['aggregate_name']] = call_user_func_array($args['aggregate_callback'],array_merge($val['a'],array($args['aggregate_arg'])));
  382.  
  383.             # original values go here, can be used e.g. by filtering
  384.             $data[$num]['aggr'] = $val['a'];
  385.         }
  386.     }
  387. #    print "keys[y]="; print_r($keys['y']);
  388. #    print "keys[x]="; print_r($keys['x']);
  389. #    print "data="; print_r($data);
  390.  
  391.     # let's sort the X axis in a natural order
  392.     # TODO: some data types should be sorted differently
  393.     uksort( $keys['x'], 'strnatcasecmp' );
  394.  
  395.     # keys for X / Y and their values (in hashes)
  396.     $val_y = $keys['y'];
  397.     $val_x = $keys['x'];
  398.  
  399.     # data column names
  400.     $val_fields = array_keys($data[0]['data']);
  401.  
  402. #    print "data=";print_r($data);
  403. #    print "val_y="; print_r($val_y);
